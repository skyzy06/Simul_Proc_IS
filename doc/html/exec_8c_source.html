29 bool decode_execute(Machine *pmach, Instruction instr) {
30  unsigned addr = pmach->_pc;
31  Code_Op op = instr.instr_generic._cop;
32  switch (op) {
33  case ILLOP: error(ERR_ILLEGAL, addr);
34  case NOP: return true; // ne fait rien
35  case LOAD: return load(pmach, instr, addr);
36  case STORE: return store(pmach, instr, addr);
37  case ADD: return add(pmach, instr, addr);

40  case CALL: return call(pmach, instr, addr);

42  case PUSH: return push(pmach, instr, addr);
43  case POP: return pop(pmach, instr, addr);
44  case HALT: return false; // arrêt normal de l'exécution

110 bool condition_respected(Machine *pmach, Instruction instr, unsigned addr) {
111  switch (instr.instr_generic._regcond) {
112  case NC: return true; // Pas de condition, toujours vraie
113  case EQ: return (pmach->_cc == CC_Z); // Egal à 0, CC == Z
114  case NE: return (pmach->_cc != CC_Z); // Différent de z, CC != Z
115  case GT: return (pmach->_cc == CC_P); // Strictement positif, CC > Z
116  case GE: return (pmach->_cc == CC_P || pmach->_cc == CC_Z); // Positif ou nul, CC >= Z
117  case LT: return (pmach->_cc == CC_N); // Strictement négatif, CC < Z
118  case LE: return (pmach->_cc == CC_N || pmach->_cc == CC_Z); // Négatif ou nul, cc <= Z
119  default: error(ERR_CONDITION, addr); // condition illégale
120  }
121 }

145 bool load(Machine *pmach, Instruction instr, unsigned addr) {
146  if (instr.instr_generic._immediate) { // si I = 1, immédiat
147  pmach->_registers[instr.instr_generic._regcond] = instr.instr_immediate._value; // R <- Val

153  //on met à jour le code condition
154  refresh_code_cond(pmach, pmach->_registers[instr.instr_generic._regcond]);
